function findOneKey(obj, key, path){
    if(typeof obj !== 'object') return ''
    if(obj[key]) return path+'.'+key
    let keys = Object.keys(obj)
    
    return [].concat(keys.map(k=>{
        return findOneKey(obj[k], key, path+'.'+k)
    }).filter(a=>!!a))
}
